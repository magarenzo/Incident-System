CREATE VIEW report AS SELECT incident.incident_id as incidentid, incident.date_created as incidentdatecreated, incident_state as incidentstate, type_body as incidenttypeinfo, comment_id as commentid, comment.date_created AS commentdatecreated, comment_body as comment, handler_name as handler FROM incident INNER JOIN incident_type on incident.incident_id = incident_type.incident_id INNER JOIN comment ON incident.incident_id = comment.incident_id INNER JOIN comment_text ON incident.incident_id = comment_text.incident_id ORDER BY comment.date_created DESC;
